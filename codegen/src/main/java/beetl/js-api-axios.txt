<%for(module in modules){%>
function ${module.moduleName}(){
    var service=new Object();
    service.url=host+"${serviceUrl}"+"${module.moduleUrl}";
    <%for(interf in module.interfaceInfos){%>
    /**
      *${interf.comment!"no comment for this interface"}
      <%for(parameter in interf.parameterInfos){%>
      *@param ${parameter.name} ${parameter.comment!"no comment"}
      <%}%>
      */
     service.${interf.methodName}=function(<%for(parameter in interf.parameterInfos){%>${parameter.name},<%}%>onSuccess){
         <%for(parameter in interf.parameterInfos){%>
	     var ${parameter.name}Str=<%if(parameter.isPrimitive){%>${parameter.name}<%}else{%>JSON.stringify(${parameter.name})<%}%>;
		 <%}%>
		 var url=this.url+"${interf.methodUrl}";
		 var parameters = {
		          <%for(parameter in interf.parameterInfos){%>
                   ${parameter.name}:${parameter.name}Str,
                   <%}%>
		 
		 }
         //当形参和实参数量相等时，执行异步的请求
        if(arguments.callee.length === arguments.length){
            axios({
                method: '${interf.requestType}',
                url: url,
                changeOrigin: true, //是否跨域
                withCredentials: true,
                <%if(interf.requestType=="Post" && !interf.hasRequestBody){%>
                data:Qs.stringify(parameters)
                <%}else{%>
                params:parameters
                <%}%>
            })
                .then(onSuccess)
                .catch(common.hngdApiRespError);
        }else{
            // common.hngdApiArgumentsError();
        }
        return{
            url:url,
            parameters:parameters
        }
    };
           
    <%}%>
    return service;
}
